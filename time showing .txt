#!/bin/bash
# Script to print current date, time, and logged-in username every 5 seconds

while true
do
    echo "--------------------------------------"
    echo "Date and Time: $(date)"
    echo "Logged-in User: $USER"
    echo "--------------------------------------"
    sleep 5
done
