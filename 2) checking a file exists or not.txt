#!/bin/bash
# Script to check if a file exists

# Ask the user for a filename
echo "Enter the file name:"
read filename

# Check if file exists
if [ -e "$filename" ]; then
    echo "✅ The file '$filename' exists."
else
    echo "❌ The file '$filename' does not exist."
fi
